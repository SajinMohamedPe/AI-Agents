# Code to calculate the first 10,000 terms of the series and multiply by 4

# Initialize variables
total = 0.0
n_terms = 10000

# Calculate the series
for n in range(n_terms):
    total += ((-1) ** n) / (2 * n + 1)

# Multiply the total by 4
result = total * 4

print(result)
